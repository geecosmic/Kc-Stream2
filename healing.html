וּבְחִבּוּרָא קַדְמָאָה פָּתַח רַעְיָא מְהֵימָנָא וְאָמַר וַוי לוֹן לִבְנֵי נָשָׁא
דְּאִינּוּן אֲטִימִין לִבָּא סְתִימִין עַיְינִין דְּלָא יַדְעִין אֵבָרִים דְּגוּפֵיהוֹן
עַל מַה אִינּוּן מִתְתַּקְּנִין דְּהָא קָנֶ״ה תְּלַת חֵילִין כְּלִילָן בֵּיהּ חַד הֶבֶל
דְּאִיהוּ לַהַב אֵשׁ דְּנָפִיק מִן לִבָּא וְאִתְפְּלַג לְז׳ הֲבָלִים דְּאָמַר קֹהֶלֶת
תִּנְיָינָא אַוִּיר דְּעָאל לְגַבֵּיהּ מִלְּבַר תְּלִיתָאָה מַיִם דְּכַנְפֵי רֵיאָה
דְּאִינּוּן דְּבוּקִים בְּקָנֶה וּמִתְּלַת אִלֵּין אִתְעַבִיד קוֹל מַיִם וְרוּחַ וְאֵשׁ
וּמִתְפְּלֵג כָּל חַד לְז׳ וְאִינּוּן ז׳ לְהָבִים ז׳ אֲוִירוֹת ז׳ נְחָלִים
